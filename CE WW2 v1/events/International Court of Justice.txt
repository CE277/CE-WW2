#Return of Territories Giver
country_event = {
	id = 1412160
	title = "Return of $FROMCOUNTRY$'s former territories?"
	desc = "EVTDESC1412160"
	picture = "territorial_negotiations"

	is_triggered_only = yes

	option = {
		name = "Return the territories!"
		prestige = 0.1
		badboy = -10
		relation = { who = FROM value = 100 }
		diplomatic_influence = { who = FROM value = 100 }
		any_owned = { limit = { is_core = FROM	NOT = { is_core = THIS } } secede_province = FROM }
		FROM = { country_event = 1412161 }
		ai_chance = { factor = 1 
			modifier = { 
				factor = 0 
				NOT = { OR = { ruling_party_ideology = socialist ruling_party_ideology = liberal } } 
			}
			modifier = { 
				factor = 0 
				AND = { is_greater_power = yes NOT = { badboy = 1 } }
				AND = { is_secondary_power = yes NOT = { badboy = 0.75 } }
			}
			modifier = { factor = 0 any_owned_province = { is_core = THIS is_core = FROM } }
			modifier = { factor = 0.1 OR = { ruling_party_ideology = fascist ruling_party_ideology = reactionary ruling_party_ideology = conservative } }
			modifier = { factor = 0 NOT = { OR = { government = democracy government = hms_government } } } }
	}

	option = {
		name = "Hold onto the territories!"
		relation = { who = FROM value = -100 }
		FROM = { country_event = 1412162 }
		ai_chance = { factor = 1 
			modifier = { factor = 2 OR = { ruling_party_ideology = fascist ruling_party_ideology = reactionary ruling_party_ideology = conservative } }
		}
	}

}

country_event = {
	id = 1512011
	title = "$COUNTRY_ADJ$ Territorial Dispute Result"
	desc = "EVTDESC1512011"
	picture = "icj_territorial_dispute"

	trigger = {
		has_country_flag = icj_in_process
		AND = { 
			THIS = { 
				is_greater_power = yes
			}
			any_neighbor_country = {
				has_country_flag = icj_marker
				is_greater_power = no
			}
		}
		NOT = { all_core = { owned_by = THIS } }
		NOT = { has_country_modifier = icj_in_process }
	}

	mean_time_to_happen =  {
		days = 2
	}

	option = {
		name = "Let's celebrate!"
		prestige = 100
		clr_country_flag = icj_in_process
		random_country = { 
			limit = { 
			has_country_flag = international_court_of_justice 
			has_country_flag = icj_marker
			is_vassal = no 
			war = no 
			any_owned_province = { is_capital = no is_core = THIS } 
			NOT = { any_owned_province = { is_capital = yes is_core = THIS } }
			neighbour = THIS 
			} 
		clr_country_flag = icj_marker
		country_event = { id = 1512017 days = 10 } 
		}
	}

}

country_event = {
	id = 1512017
	title = "International Court of Justice rules against us"
	desc = "EVTDESC1512017"
	picture = "icj_territorial_dispute"

	is_triggered_only = yes

	option = {
		name = "Accept the ruling"
		prestige_factor = 0.05
		badboy = -10
		FROM = { relation = { who = THIS value = 100 } }
		clr_country_flag = icj_marker
		any_owned = { limit = { is_capital = no is_core = FROM } remove_core = THIS secede_province = FROM }
		ai_chance = { 
			factor = 1 
			modifier = { factor = 0 NOT = { OR = { ruling_party_ideology = liberal ruling_party_ideology = socialist } } }
		}
	}

	option = {
		name = "Ignore the ruling!"
		prestige_factor = -0.05
		badboy = 15
		FROM = { relation = { who = THIS value = -100 } country_event = { id = 1512018 days = 10 } }
		any_owned = { limit = { is_core = FROM } add_province_modifier = { name = nationalist_agitation duration = 3650 } }
		clr_country_flag = icj_marker
		ai_chance = { 
			factor = 1 
			modifier = { factor = 0 OR = { ruling_party_ideology = liberal ruling_party_ideology = socialist } }
		}
	}

}
