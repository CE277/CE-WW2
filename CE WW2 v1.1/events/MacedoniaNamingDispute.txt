#Prespa Agreement
country_event = {
	id = 20014
	title = "EVTNAME20014"
	desc = "EVTDESC20014"
	picture = "prespa_agreement"
	
	trigger = {
		tag = GRE
		exists = MCD
		MCD = { has_country_flag = open_to_negotiations }
		has_country_flag = gre_fyrom_negotiations
		NOT = { has_country_flag = sign_prespa_agreement }
		NOT = { has_country_flag = refuse_prespa_agreement }
	}
		
	mean_time_to_happen = { months = 24 }

	fire_only_once = yes
	
	option = {
		name = "Sign the agreement."
		set_country_flag = sign_prespa_agreement
		prestige = 20
		any_pop = {
			scaled_militancy = { ideology = conservative factor = 2 }
			scaled_militancy = { ideology = reactionary factor = 3 }
			scaled_militancy = { ideology = fascist factor = 4 }
		}
		relation = { who = MCD value = 75 }
		MCD = { country_event = 20015 } #Prespa Agreement - MCD
		ai_chance = {
			factor = 1
		}
	}

	option = {
		name = "Refuse to sign the agreement."
		set_country_flag = refuse_prespa_agreement
		prestige = -75
		any_pop = {
			scaled_militancy = { ideology = communist factor = 2 }
			scaled_militancy = { ideology = socialist factor = 3 }
		}
		relation = { who = MCD value = -100 }
		ai_chance = {
			factor = 0
		}
	}

}
